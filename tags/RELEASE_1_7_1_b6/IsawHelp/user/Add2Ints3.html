from DataSetTools.operator import *

from Command import ScriptUtil
from DataSetTools.util import *
from gov.anl.ipns.Util.Sys import *;

class Add2Ints3(GenericOperator):
    def setDefaultParameters(self):
        self.super__clearParametersVector()
        self.addParameter( IntegerPG("First Number",Integer(5)))
        self.addParameter( IntegerPG("Second Number", Integer(9)))
        
        
    def getResult(self):
        FirstNum = self.getParameter(0).getintValue()
        SecondNum = self.getParameter(1).getintValue()
        Result = FirstNum + SecondNum
        return Result


    def getDocumentation(self):
        S = "@overview Adds two integers"
        S += " @param  FirstNum  The first number to add"
        S += " @param  SecondNum   The second number to add"
        S += "@return The sum of the two numbers"
 
        return S

    def __init__(self):
        Operator.__init__(self,"Add 2 Ints 3")
